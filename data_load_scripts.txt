// -----------------------------
// LINK TABLE FOR DIMENSIONS (prevents synthetic keys)
// Facts will connect ONLY via LinkKey
// -----------------------------
LinkFacts:
LOAD
  ClaimCaseKey as LinkKey,
  PolicyKey,
  OrgKey,
  PlanKey,
  AgentKey,
  ClaimType,
  PaymentType,
  DeathClaimType,
  ClaimStatus
RESIDENT FactClaimsLifecycle_Raw;

CONCATENATE (LinkFacts)
LOAD
  'D' & DueCaseKey as LinkKey,
  PolicyKey,
  OrgKey,
  PlanKey,
  AgentKey,
  ClaimType,
  PaymentType,
  '' as DeathClaimType,
  '' as ClaimStatus
RESIDENT FactClaimsDue_Raw;


// -----------------------------
// LINK TABLE FOR MULTIPLE DATES (Booked/Paid/Due/Intimation)
// -----------------------------
LinkDates:
LOAD
  ClaimCaseKey as LinkKey,
  ClaimBookedDateKey as DateKey,
  'Booked' as DateRole
RESIDENT FactClaimsLifecycle_Raw
WHERE Len(ClaimBookedDateKey) > 0;

CONCATENATE (LinkDates)
LOAD
  ClaimCaseKey as LinkKey,
  ClaimPaidDateKey as DateKey,
  'Paid' as DateRole
RESIDENT FactClaimsLifecycle_Raw
WHERE Len(ClaimPaidDateKey) > 0;

CONCATENATE (LinkDates)
LOAD
  ClaimCaseKey as LinkKey,
  ClaimPaymentDueDateKey as DateKey,
  'Due' as DateRole
RESIDENT FactClaimsLifecycle_Raw
WHERE Len(ClaimPaymentDueDateKey) > 0;

CONCATENATE (LinkDates)
LOAD
  ClaimCaseKey as LinkKey,
  ClaimIntimationDateKey as DateKey,
  'Intimation' as DateRole
RESIDENT FactClaimsLifecycle_Raw
WHERE Len(ClaimIntimationDateKey) > 0;

CONCATENATE (LinkDates)
LOAD
  'D' & DueCaseKey as LinkKey,
  ClaimDueDateKey as DateKey,
  'Due' as DateRole
RESIDENT FactClaimsDue_Raw
WHERE Len(ClaimDueDateKey) > 0;

CONCATENATE (LinkDates)
LOAD
  'D' & DueCaseKey as LinkKey,
  IntimationDateKey as DateKey,
  'Intimation' as DateRole
RESIDENT FactClaimsDue_Raw
WHERE Len(IntimationDateKey) > 0;
